 49 -def pips(f_fcn, x0, A=None, l=None, u=None, xmin=None, xmax=None, 
 50           gh_fcn=None, hess_fcn=None, opt=None): 

 51      """Primal-dual interior point method for NLP (non-linear programming). 
 52      Minimize a function F(X) beginning from a starting point M{x0}, subject to 
 53      optional linear and non-linear constraints and variable bounds:: 
 54   
 55              min f(x) 
 56               x 
 57   
 58      subject to:: 
 59   
 60              g(x) = 0            (non-linear equalities) 
 61              h(x) <= 0           (non-linear inequalities) 
 62              l <= A*x <= u       (linear constraints) 
 63              xmin <= x <= xmax   (variable bounds) 
 64   
 65      Note: The calling syntax is almost identical to that of FMINCON from 
 66      MathWorks' Optimization Toolbox. The main difference is that the linear 
 67      constraints are specified with C{A}, C{L}, C{U} instead of C{A}, C{B}, 
 68      C{Aeq}, C{Beq}. The functions for evaluating the objective function, 
 69      constraints and Hessian are identical. 
 70   
 71      Example from U{http://en.wikipedia.org/wiki/Nonlinear_programming}: 
 72          >>> from numpy import array, r_, float64, dot 
 73          >>> from scipy.sparse import csr_matrix 
 74          >>> def f2(x): 
 75          ...     f = -x[0] * x[1] - x[1] * x[2] 
 76          ...     df = -r_[x[1], x[0] + x[2], x[1]] 
 77          ...     # actually not used since 'hess_fcn' is provided 
 78          ...     d2f = -array([[0, 1, 0], [1, 0, 1], [0, 1, 0]], float64) 
 79          ...     return f, df, d2f 
 80          >>> def gh2(x): 
 81          ...     h = dot(array([[1, -1, 1], 
 82          ...                    [1,  1, 1]]), x**2) + array([-2.0, -10.0]) 
 83          ...     dh = 2 * csr_matrix(array([[ x[0], x[0]], 
 84          ...                                [-x[1], x[1]], 
 85          ...                                [ x[2], x[2]]])) 
 86          ...     g = array([]) 
 87          ...     dg = None 
 88          ...     return h, g, dh, dg 
 89          >>> def hess2(x, lam): 
 90          ...     mu = lam["ineqnonlin"] 
 91          ...     a = r_[dot(2 * array([1, 1]), mu), -1, 0] 
 92          ...     b = r_[-1, dot(2 * array([-1, 1]),mu),-1] 
 93          ...     c = r_[0, -1, dot(2 * array([1, 1]),mu)] 
 94          ...     Lxx = csr_matrix(array([a, b, c])) 
 95          ...     return Lxx 
 96          >>> x0 = array([1, 1, 0], float64) 
 97          >>> solution = pips(f2, x0, gh_fcn=gh2, hess_fcn=hess2) 
 98          >>> round(solution["f"], 11) == -7.07106725919 
 99          True 
100          >>> solution["output"]["iterations"] 
101          8 
102   
103      Ported by Richard Lincoln from the MATLAB Interior Point Solver (MIPS) 
104      (v1.9) by Ray Zimmerman.  MIPS is distributed as part of the MATPOWER 
105      project, developed at the Power System Engineering Research Center (PSERC), 
106      Cornell. See U{http://www.pserc.cornell.edu/matpower/} for more info. 
107      MIPS was ported by Ray Zimmerman from C code written by H. Wang for his 
108      PhD dissertation: 
109        - "On the Computation and Application of Multi-period 
110          Security-Constrained Optimal Power Flow for Real-time 
111          Electricity Market Operations", Cornell University, May 2007. 
112   
113      See also: 
114        - H. Wang, C. E. Murillo-Sanchez, R. D. Zimmerman, R. J. Thomas, 
115          "On Computational Issues of Market-Based Optimal Power Flow", 
116          IEEE Transactions on Power Systems, Vol. 22, No. 3, Aug. 2007, 
117          pp. 1185-1193. 
118   
119      All parameters are optional except C{f_fcn} and C{x0}. 
120      @param f_fcn: Function that evaluates the objective function, its gradients 
121                    and Hessian for a given value of M{x}. If there are 
122                    non-linear constraints, the Hessian information is provided 
123                    by the 'hess_fcn' argument and is not required here. 
124      @type f_fcn: callable 
125      @param x0: Starting value of optimization vector M{x}. 
126      @type x0: array 
127      @param A: Optional linear constraints. 
128      @type A: csr_matrix 
129      @param l: Optional linear constraints. Default values are M{-Inf}. 
130      @type l: array 
131      @param u: Optional linear constraints. Default values are M{Inf}. 
132      @type u: array 
133      @param xmin: Optional lower bounds on the M{x} variables, defaults are 
134                   M{-Inf}. 
135      @type xmin: array 
136      @param xmax: Optional upper bounds on the M{x} variables, defaults are 
137                   M{Inf}. 
138      @type xmax: array 
139      @param gh_fcn: Function that evaluates the optional non-linear constraints 
140                     and their gradients for a given value of M{x}. 
141      @type gh_fcn: callable 
142      @param hess_fcn: Handle to function that computes the Hessian of the 
143                       Lagrangian for given values of M{x}, M{lambda} and M{mu}, 
144                       where M{lambda} and M{mu} are the multipliers on the 
145                       equality and inequality constraints, M{g} and M{h}, 
146                       respectively. 
147      @type hess_fcn: callable 
148      @param opt: optional options dictionary with the following keys, all of 
149                  which are also optional (default values shown in parentheses) 
150                    - C{verbose} (False) - Controls level of progress output 
151                      displayed 
152                    - C{feastol} (1e-6) - termination tolerance for feasibility 
153                      condition 
154                    - C{gradtol} (1e-6) - termination tolerance for gradient 
155                      condition 
156                    - C{comptol} (1e-6) - termination tolerance for 
157                      complementarity condition 
158                    - C{costtol} (1e-6) - termination tolerance for cost 
159                      condition 
160                    - C{max_it} (150) - maximum number of iterations 
161                    - C{step_control} (False) - set to True to enable step-size 
162                      control 
163                    - C{max_red} (20) - maximum number of step-size reductions if 
164                      step-control is on 
165                    - C{cost_mult} (1.0) - cost multiplier used to scale the 
166                      objective function for improved conditioning. Note: The 
167                      same value must also be passed to the Hessian evaluation 
168                      function so that it can appropriately scale the objective 
169                      function term in the Hessian of the Lagrangian. 
170      @type opt: dict 
171   
172      @rtype: dict 
173      @return: The solution dictionary has the following keys: 
174                 - C{x} - solution vector 
175                 - C{f} - final objective function value 
176                 - C{converged} - exit status 
177                     - True = first order optimality conditions satisfied 
178                     - False = maximum number of iterations reached 
179                     - None = numerically failed 
180                 - C{output} - output dictionary with keys: 
181                     - C{iterations} - number of iterations performed 
182                     - C{hist} - dictionary of arrays with trajectories of the 
183                       following: feascond, gradcond, compcond, costcond, gamma, 
184                       stepsize, obj, alphap, alphad 
185                     - C{message} - exit message 
186                 - C{lmbda} - dictionary containing the Langrange and Kuhn-Tucker 
187                   multipliers on the constraints, with keys: 
188                     - C{eqnonlin} - non-linear equality constraints 
189                     - C{ineqnonlin} - non-linear inequality constraints 
190                     - C{mu_l} - lower (left-hand) limit on linear constraints 
191                     - C{mu_u} - upper (right-hand) limit on linear constraints 
192                     - C{lower} - lower bound on optimization variables 
193                     - C{upper} - upper bound on optimization variables 
194   
195      @license: Apache License version 2.0 
196      """ 
197      nx = x0.shape[0]                        # number of variables 
198      nA = A.shape[0] if A is not None else 0 # number of original linear constr 
199   
200      # default argument values 
201  #    l = array([]) if A is None else l 
202  #    u = array([]) if A is None else u 
203      l = -Inf * ones(nA) if l is None else l 
204      u =  Inf * ones(nA) if u is None else u 
205      xmin = -Inf * ones(x0.shape[0]) if xmin is None else xmin 
206      xmax =  Inf * ones(x0.shape[0]) if xmax is None else xmax 
207      if gh_fcn is None: 
208          nonlinear = False 
209          gn = array([]) 
210          hn = array([]) 
211      else: 
212          nonlinear = True 
213   
214      opt = {} if opt is None else opt 
215      # options 
216      if not opt.has_key("feastol"): 
217          opt["feastol"] = 1e-06 
218      if not opt.has_key("gradtol"): 
219          opt["gradtol"] = 1e-06 
220      if not opt.has_key("comptol"): 
221          opt["comptol"] = 1e-06 
222      if not opt.has_key("costtol"): 
223          opt["costtol"] = 1e-06 
224      if not opt.has_key("max_it"): 
225          opt["max_it"] = 150 
226      if not opt.has_key("max_red"): 
227          opt["max_red"] = 20 
228      if not opt.has_key("step_control"): 
229          opt["step_control"] = False 
230      if not opt.has_key("cost_mult"): 
231          opt["cost_mult"] = 1 
232      if not opt.has_key("verbose"): 
233          opt["verbose"] = False 
234   
235      # initialize history 
236      hist = {} 
237   
238      # constants 
239      xi = 0.99995 
240      sigma = 0.1 
241      z0 = 1 
242      alpha_min = 1e-8 
243  #    rho_min = 0.95 
244  #    rho_max = 1.05 
245      mu_threshold = 1e-5 
246   
247      # initialize 
248      i = 0                       # iteration counter 
249      converged = False           # flag 
250      eflag = False               # exit flag 
251   
252      # add var limits to linear constraints 
253      eyex = eye(nx, nx, format="csr") 
254      AA = eyex if A is None else vstack([eyex, A], "csr") 
255      ll = r_[xmin, l] 
256      uu = r_[xmax, u] 
257   
258      # split up linear constraints 
259      ieq = flatnonzero( absolute(uu - ll) <= EPS ) 
260      igt = flatnonzero( (uu >=  1e10) & (ll > -1e10) ) 
261      ilt = flatnonzero( (ll <= -1e10) & (uu <  1e10) ) 
262      ibx = flatnonzero( (absolute(uu - ll) > EPS) & (uu < 1e10) & (ll > -1e10) ) 
263      # zero-sized sparse matrices unsupported 
264      Ae = AA[ieq, :] if len(ieq) else None 
265      if len(ilt) or len(igt) or len(ibx): 
266          idxs = [(1, ilt), (-1, igt), (1, ibx), (-1, ibx)] 
267          Ai = vstack([sig * AA[idx, :] for sig, idx in idxs if len(idx)]) 
268      else: 
269          Ai = None 
270      be = uu[ieq, :] 
271      bi = r_[uu[ilt], -ll[igt], uu[ibx], -ll[ibx]] 
272   
273      # evaluate cost f(x0) and constraints g(x0), h(x0) 
274      x = x0 
275      f, df, _ = f_fcn(x)                 # cost 
276      f = f * opt["cost_mult"] 
277      df = df * opt["cost_mult"] 
278      if nonlinear: 
279          hn, gn, dhn, dgn = gh_fcn(x)        # non-linear constraints 
280          h = hn if Ai is None else r_[hn, Ai * x - bi] # inequality constraints 
281          g = gn if Ae is None else r_[gn, Ae * x - be] # equality constraints 
282   
283          if (dhn is None) and (Ai is None): 
284              dh = None 
285          elif dhn is None: 
286              dh = Ai.T 
287          elif Ae is None: 
288              dh = dhn 
289          else: 
290              dh = hstack([dhn, Ai.T]) 
291   
292          if (dgn is None) and (Ae is None): 
293              dg = None 
294          elif dgn is None: 
295              dg = Ae.T 
296          elif Ae is None: 
297              dg = dgn 
298          else: 
299              dg = hstack([dgn, Ae.T]) 
300      else: 
301          h = -bi if Ai is None else Ai * x - bi        # inequality constraints 
302          g = -be if Ae is None else Ae * x - be        # equality constraints 
303          dh = None if Ai is None else Ai.T     # 1st derivative of inequalities 
304          dg = None if Ae is None else Ae.T     # 1st derivative of equalities 
305   
306      # some dimensions 
307      neq = g.shape[0]           # number of equality constraints 
308      niq = h.shape[0]           # number of inequality constraints 
309      neqnln = gn.shape[0]       # number of non-linear equality constraints 
310      niqnln = hn.shape[0]       # number of non-linear inequality constraints 
311      nlt = len(ilt)             # number of upper bounded linear inequalities 
312      ngt = len(igt)             # number of lower bounded linear inequalities 
313      nbx = len(ibx)             # number of doubly bounded linear inequalities 
314   
315      # initialize gamma, lam, mu, z, e 
316      gamma = 1                  # barrier coefficient 
317      lam = zeros(neq) 
318      z = z0 * ones(niq) 
319      mu = z0 * ones(niq) 
320      k = flatnonzero(h < -z0) 
321      z[k] = -h[k] 
322      k = flatnonzero((gamma / z) > z0) 
323      mu[k] = gamma / z[k] 
324      e = ones(niq) 
325   
326      # check tolerance 
327      f0 = f 
328  #    if opt["step_control"]: 
329  #        L = f + lam.T * g + mu.T * (h + z) - gamma * sum(log(z)) 
330   
331      Lx = df 
332      Lx = Lx + dg * lam if dg is not None else Lx 
333      Lx = Lx + dh * mu  if dh is not None else Lx 
334   
335      gnorm = norm(g, Inf) if len(g) else 0.0 
336      lam_norm = norm(lam, Inf) if len(lam) else 0.0 
337      mu_norm = norm(mu, Inf) if len(mu) else 0.0 
338      feascond = \ 
339          max([gnorm, max(h)]) / (1 + max([norm(x, Inf), norm(z, Inf)])) 
340      gradcond = \ 
341          norm(Lx, Inf) / (1 + max([lam_norm, mu_norm])) 
342      compcond = dot(z, mu) / (1 + norm(x, Inf)) 
343      costcond = absolute(f - f0) / (1 + absolute(f0)) 
344   
345      # save history 
346      hist[i] = {'feascond': feascond, 'gradcond': gradcond, 
347          'compcond': compcond, 'costcond': costcond, 'gamma': gamma, 
348          'stepsize': 0, 'obj': f / opt["cost_mult"], 'alphap': 0, 'alphad': 0} 
349   
350      if opt["verbose"]: 
351  #        s = '-sc' if opt["step_control"] else '' 
352  #        version, date = '1.0b2', '24-Mar-2010' 
353  #        print 'Python Interior Point Solver - PIPS%s, Version %s, %s' % \ 
354  #                    (s, version, date) 
355          print " it    objective   step size   feascond     gradcond     " \ 
356                "compcond     costcond  " 
357          print "----  ------------ --------- ------------ ------------ " \ 
358                "------------ ------------" 
359          print "%3d  %12.8g %10s %12g %12g %12g %12g" % \ 
360              (i, (f / opt["cost_mult"]), "", 
361               feascond, gradcond, compcond, costcond) 
362   
363      if feascond < opt["feastol"] and gradcond < opt["gradtol"] and \ 
364          compcond < opt["comptol"] and costcond < opt["costtol"]: 
365          converged = True 
366          if opt["verbose"]: 
367              print "Converged!" 
368   
369      # do Newton iterations 
370      while (not converged and i < opt["max_it"]): 
371          # update iteration counter 
372          i += 1 
373   
374          # compute update step 
375          lmbda = {"eqnonlin": lam[range(neqnln)], 
376                   "ineqnonlin": mu[range(niqnln)]} 
377          if nonlinear: 
378              if hess_fcn is None: 
379                  print "pips: Hessian evaluation via finite differences " \ 
380                        "not yet implemented.\nPlease provide " \ 
381                        "your own hessian evaluation function." 
382              Lxx = hess_fcn(x, lmbda) 
383          else: 
384              _, _, d2f = f_fcn(x)      # cost 
385              Lxx = d2f * opt["cost_mult"] 
386          rz = range(len(z)) 
387          zinvdiag = csr_matrix((1.0 / z, (rz, rz))) if len(z) else None 
388          rmu = range(len(mu)) 
389          mudiag = csr_matrix((mu, (rmu, rmu))) if len(mu) else None 
390          dh_zinv = None if dh is None else dh * zinvdiag 
391          M = Lxx if dh is None else Lxx + dh_zinv * mudiag * dh.T 
392          N = Lx if dh is None else Lx + dh_zinv * (mudiag * h + gamma * e) 
393   
394          Ab = M if dg is None else vstack([ 
395              hstack([M, dg]), 
396              hstack([dg.T, csr_matrix((neq, neq))]) 
397          ]) 
398          bb = r_[-N, -g] 
399   
400          dxdlam = spsolve(Ab.tocsr(), bb) 
401   
402          dx = dxdlam[:nx] 
403          dlam = dxdlam[nx:nx + neq] 
404          dz = -h - z if dh is None else -h - z - dh.T * dx 
405          dmu = -mu if dh is None else -mu + zinvdiag * (gamma * e - mudiag * dz) 
406   
407          # optional step-size control 
408  #        sc = False 
409          if opt["step_control"]: 
410              raise NotImplementedError

455          # do the update 
456          k = flatnonzero(dz < 0.0) 
457          alphap = min([xi * min(z[k] / -dz[k]), 1]) if len(k) else 1.0 
458          k = flatnonzero(dmu < 0.0) 
459          alphad = min([xi * min(mu[k] / -dmu[k]), 1]) if len(k) else 1.0

461          z = z + alphap * dz

463          mu = mu + alphad * dmu

469          f = f * opt["cost_mult"] 
470          df = df * opt["cost_mult"] 
471          if nonlinear:

473  #            g = gn if Ai is None else r_[gn, Ai * x - bi] # ieq constraints 
474  #            h = hn if Ae is None else r_[hn, Ae * x - be] # eq constraints 
475              h = hn if Ai is None else r_[hn, Ai * x - bi] # ieq constr 
476              g = gn if Ae is None else r_[gn, Ae * x - be]  # eq constr 
477   
478              if (dhn is None) and (Ai is None): 
479                  dh = None 
480              elif dhn is None: 
481                  dh = Ai.T 
482              elif Ae is None: 
483                  dh = dhn 
484              else:

488                  dg = None 
489              elif dgn is None: 
490                  dg = Ae.T 
491              elif Ae is None: 
492                  dg = dgn 
493              else: 
494                  dg = hstack([dgn, Ae.T]) 
495          else:

497              g = -be if Ae is None else Ae * x - be    # equality constraints 
498              # 1st derivatives are constant, still dh = Ai.T, dg = Ae.T 
499   
500          Lx = df 
501          Lx = Lx + dg * lam if dg is not None else Lx 
502          Lx = Lx + dh * mu  if dh is not None else Lx 
503   
504          gnorm = norm(g, Inf) if len(g) else 0.0 
505          lam_norm = norm(lam, Inf) if len(lam) else 0.0 
506          mu_norm = norm(mu, Inf) if len(mu) else 0.0

512          costcond = float(absolute(f - f0) / (1 + absolute(f0)))

525              compcond < opt["comptol"] and costcond < opt["costtol"]:

532                  if opt["verbose"]:
